Jasmine: Basics

helloWorld.js


const helloWorld = function() {
    return 'Hello, World!';
}
module.exports = helloWorld
module.exports
Exports the function so it can be imported with require()

helloWorld.spec.js


const helloWorld = require('./helloWorld');

describe('Hello World', function() {
    it('says hello world', function() {
        expect(helloWorld()).toEqual('Hello, World!');
    });
});
require()
Imports the code to be tested.
describe()
Body of the test.
it()
Describes what should be happening in plain english.
xit()
Disable test
expect()
Test condition.

Run in console:


jasmine helloWorld.spec.js
