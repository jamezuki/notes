Random


/* random number between min and max (both included) */
function getRndInteger(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

Factorial


function factorial(num) {
	if (num == 0) return 1;
	let result = 1;
	for (let i = num; i > 0; i--) {
		result = result * i;
	}
	return result;
}

Palindrome


const palindrome = function(str) {
    array1 = str.toLowerCase().split('');
    let noSpaceArray = array1.filter(n => /^[A-Z]$/i.test(n) ? true : false);
    let noSpaceStr = noSpaceArray.join('');
    let reverseStr = noSpaceArray.reverse().join('');
    return noSpaceStr === reverseStr ? true : false;
}

Fibonacci


const fibonacci = function(num) {
    if ( num < 0) return 'OOPS';
    let fibonacci = [1, 1];
    if (num > 0 && num < 3) return fibonacci[num - 1];
    for(let i = 2; i < num; i++){
        fibonacci.push(fibonacci[i-1] + fibonacci[i-2]);
    }
    return fibonacci[num - 1];
}

findTheOldest


let findTheOldest = function (people) {
    return people.reduce((oldest, person) => {
        if (!oldest.yearOfDeath) oldest.yearOfDeath = (new Date()).getFullYear();
        if (!person.yearOfDeath) person.yearOfDeath = (new Date()).getFullYear();
        if( oldest.yearOfDeath - oldest.yearOfBirth < person.yearOfDeath - person.yearOfBirth) {
            return person
        }
        return oldest;
    });
}

Form validation


const inputs = document.querySelectorAll('input');

const patterns = {
    telephone: /^\d{10}$/,
    username: /^[a-z\d]{5,12}$/i,
    password: /^[\w@-]{8,20}$/,
    email: /^([a-z\d\.-]+)@([a-z\d-]+)\.([a-z]{2,8})(\.[a-z]{2,8})?$/,
}

function validate(field, regex){
    if (regex.test(field.value)){
        field.className = 'valid';
    } else {
        field.className = 'invalid';
    }
}

inputs.forEach(input => {
    input.addEventListener('keyup', e => {
        validate(e.target, patterns[e.target.attributes.name.value]);
    });
});
