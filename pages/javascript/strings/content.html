Strings

  • Primitive data type.
  • Immutable: cannot be changed, only replaced.

let x = "Hello";
let x = 'Hello';
let x = `Hello ${y}`;

Length


let str = 'one';
str.length; // 3

Concatenate


let one = 'Hello, ';
let two = 'how are you?';
one + two; // "Hello, how are you?"

string + number = string

unicode + 0060

unicode: ctrl + shift + u


let x = 'James';
`Hello, ${x}!` // "Hello, James!"
`The result is ${1 + 2}` // "The result is 3"

Convert: String to Number

Number(string)


typeof(Number(str)); // number

Convert: Number to String

.toString()


typeof(num.toString()); // string

Convert: String to Array

.split(separator)

.split(separator, limit)


let s1 = "a,b,c,d,e";
let a1 = x.split(','); // [a, b, c, d, e]

let s2 = "Hello";
let a2 = x.split(''); // [H, e, l, l, o]

Convert: Uppercase/Lowercase

.toUpperCase()


let x = str.toUpperCase();

.toLowerCase()


let x = str.toLowerCase();

Search: Includes

.includes(search)

.includes(search, start)

  • Returns true if the search string is found

const sentence = 'The quick brown fox';

sentence.includes('fox'); // true

Search: Index of First match

.indexOf(search)

.indexOf(search, start)

  • Returns -1 if not found.

let str = "01234567895";
str.indexOf("5"); // 5
str.indexOf("5", 9); // 10

.search(regex)


str.search('/^Hello$/');

Search: Index of Last match

.lastIndexOf(search)

.lastIndexOf(search, start)

  • Searches backwards.
  • start is counted form end of string.
  • Returns -1 if not found.

'canal'.lastIndexOf('a');     // returns 3
'canal'.lastIndexOf('a', 2);  // returns 1

Extract: Strings

.slice(startIndex)

.slice(startIndex, endIndex)

  • endIndex not included in the extracted string.
  • If a parameter is negative, the position is counted from the end of the string.
  • If endIndex is omitted returns the rest of the string.

let str = "0123456789";
let x1 = str.slice(2, 4); // 23
let x2 = str.slice(0, -6); // 012
let x3 = str.slice(7); // 789

.substr(start, length)


let str = "0123456789";
let x = str.substr(2, 2); // 23

Extract: Characters

.charAt(index)


let str = "0123456789";
str.charAt(5); // 5

str[index]

  • Strings are not arrays.
  • Read only (strings are immutable).
  • Returns undefined if not found.

let str = "0123456789";
str[5]; // 5

Replace

.replace(oldStr, newStr)

  • Case sensitive.
  • Replaces only the first match.
  • Returns a new string.
  • Regex are written without quotes.

let str1 = "Apple, Banana, Kiwi";
let str2 = str.replace("Banana", "Melon");
// "Apple, Melon, Kiwi"

Case insensitive regex


let str3 = str.replace(/BANANA/i, "Melon");

Replace all regex:


let str4 = str.replace(/Banana/g, "Melon");

Regex

Example


// 10 digits from start to end
let telephone = /^\d{10}$/;
let str = '0123456789';

telephone.test('str'); // true

Basics


/foo/       // first 'foo' match
/foo/g      // multiple matches
/foo/i      // case insensitive

Ranges


/[a-z]/       // every letter of alphabet

Character Sets


/[ab]/      // first letter must be 'a' or 'b'
/[^p]/      // first letter must be any character except 'p'

Repeating Characters


/[0-9]+/        // at least 1 number from 0 to 9
/[0-9]{11}/     // 11 numbers between 0 to 9
/[0-9]{5,9}/    // 5 to 9 numbers between 0 to 9
/[0-9]{5,}/     // at least 5 numbers between 0 to 9

Meta-characters


\d      // any digit, same as [0-9]
\w      // any word character a-z, A-Z, 0-9 and _
\s      // any whitespace characer (space, tab, etc)
\t      // tab character only

Special characters


/\./        // match '.'
/foo?/      // last 'o' is 0 or 1 (is optional)
/foo*/      // last 'o' is >= 0
/./         // any character except new line

Starting & Ending Patterns


/^foo/      // must start with 'foo'
/foo$/      // must end with 'foo'
/^foo$/     // only 'foo' (start & end)

Alternate Characters


/(F|f)oo/       // 'Foo' or 'foo'
